Merge CBS data en -9997 en -9995 op NA zetten.

cbs_pc6 <- bind_rows(cbs_pc6_24, cbs_pc6_23, cbs_pc6_22) |>
  arrange (postcode6, year) |>
  relocate (postcode6, year, geom)|>
  mutate(across(where(is.numeric), ~na_if(., -99997))) |>
  mutate(across(where(is.numeric), ~na_if(., -99995)))

Een aantal van de straten in de PV data hebben meerdere postcodes:

pv <- readRDS("./data/pv_data.Rds") |>
  filter(year >= 2022) |>
  mutate(id = row_number()) |>
  relocate(id)

meerdere_postcodes <- pv |> 
  mutate(same_pc = if_else(POSTCODE_VAN == POSTCODE_TOT, 1, 0)) |>
  filter(same_pc == 0) |>
  distinct(STRAATNAAM, POSTCODE_VAN, POSTCODE_TOT, .keep_all = TRUE)
