1 First look at CBS data

After having assembled the solar panel data, it is time to connect it to the demographic characteristics of the area.

First, we load the relevant packages

rm(list=ls())

library(tidyverse)
library(sf)

Most of the data we want is known at the postal code level, so we import the CBS data per PC6 postal code. These data can be downloaded here: https://www.cbs.nl/nl-nl/dossier/nederland-regionaal/geografische-data/gegevens-per-postcode. The data from 2020 and 2019 is unavailable at this moment but should be available before the 1st of May. The data from 2021-2024 and 2015-2018 are named a bit differently so we first rename and select the relevant columns before joining.

Some streets in the PV data have multiple postal codes:

#load pv data and add id
pv <- readRDS("./data/pv_data.Rds") |>
  filter(year >= 2015) |>
  mutate(id = row_number()) |>
  relocate(id)

#check for multiple postal codes per street
meerdere_postcodes <- pv |> 
  mutate(same_pc = if_else(POSTCODE_VAN == POSTCODE_TOT, 1, 0)) |>
  filter(same_pc == 0) |>
  distinct(STRAATNAAM, POSTCODE_VAN, POSTCODE_TOT, .keep_all = TRUE)

We select the CBS postal codes which match the streets from the PV data.

cbs_pc6_westland <- cbs_pc6 |>
  inner_join(
    pv, 
    by = join_by(
      year == year,
      postcode6 >= POSTCODE_VAN,
      postcode6 <= POSTCODE_TOT
    )
  ) |>
  relocate(postcode6, geom, year, id, POSTCODE_VAN, POSTCODE_TOT, STRAATNAAM, X.Leveringsrichting)

rm(cbs_pc6, meerdere_postcodes)
gc()

Now we join the squares with the postal code areas based on the largest overlapping area.

#match the squares belonging to the postal code area and keep only the square coordinates
df <- df |>
  st_join(cbs_500,join = st_intersects, left = T, largest = T) |>
  select(-c(year.y, gemiddeld_inkomen_huishouden, percentage_laag_inkomen_huishouden, percentage_hoog_inkomen_huishouden, omgevingsadressendichtheid, stedelijkheid, M_INKHH))

#match the square attributes in the right year
df <- df |>
  left_join (st_drop_geometry(cbs_500), by = c("crs28992res500m", "year.x" = "year")) |>
  rename(year = year.x) |>
  mutate(across(where(is.numeric), ~na_if(., NaN)))
